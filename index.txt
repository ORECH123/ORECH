let pijWojankaPlayed = false;

function openDuckWindow() {
  const width = 480;
  const height = 270;
  let left = Math.floor(Math.random() * (window.screen.width - width));
  let top = Math.floor(Math.random() * (window.screen.height - height));
  let dx = Math.random() < 0.5 ? 4 : -4;
  let dy = Math.random() < 0.5 ? 4 : -4;

  // otwieramy prawdziwe nowe okno
  const win = window.open("duck.html", "_blank", `width=${width},height=${height},left=${left},top=${top}`);
  if (!win) return;

  // animacja "latajÄ…cego" okna
  function move() {
    if (win.closed) return;
    left += dx;
    top += dy;
    if (left <= 0 || left + width >= window.screen.width) dx = -dx;
    if (top <= 0 || top + height >= window.screen.height) dy = -dy;
    win.moveTo(left, top);
    requestAnimationFrame(move);
  }
  move();
}

document.querySelectorAll('.plan').forEach(plan => {
  plan.addEventListener('click', () => {
    openDuckWindow();

    if (!pijWojankaPlayed) {
      pijWojankaPlayed = true;
      setTimeout(() => {
        const audio = new Audio('pij_wojanka.mp3');
        audio.volume = 1.0;
        audio.play().catch(() => {
          document.body.addEventListener('click', () => audio.play(), { once: true });
        });
      }, 5000);
    }
  });
});
